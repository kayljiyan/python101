````md magic-move {lines: true}
```python
# Arithmetic Operators
a = 9
b = 3
add = a + b
sub = a - b
mul = a * b
div = a / b
mod = a % b
power = a ** b
print(add) # 12
print(sub) # 6
print(mul) # 27
print(div) # 3
print(mod) # 0
print(power) # 729
```
```python
# Logical Operators
a = True
b = False
print(a and b) # False
print(a or b) # True
print(not a) # False
num1 = 5
num2 = 3
num3 = 5
print(num1 == num2) # False
print(num1 > num2) # True
print(num1 < num2) # False
print(num1 >= num2) # True
print(num1 <= num3) # True
```
```python
# Assignment Operators
a = 10
b = a
print(b) # 10
b += a # b = b + a
print(b) # 20
b -= a # b = b - a
print(b) # 10
b *= a # b = b * a
print(b) # 100
````
